#def the plot confusion matrix funtion

def plot_confusion_matrix(cm, title): 
    plt.figure(figsize = (20,17))
    plt.imshow(cm, interpolation='nearest',cmap=plt.cm.Reds)    #     
    plt.title(title)    #     
    plt.colorbar()     
    plt.ylabel('True label')        
    plt.xlabel('Predicted label')
    indices = range(len(cm))
    plt.xticks(indices, ["T-shirt/top", "Trouser", "Pullover", "Dress", "Coat", "Sandal", "Shirt", "Sneaker", "Bag", "Ankle boot"])
    plt.yticks(indices, ["T-shirt/top", "Trouser", "Pullover", "Dress", "Coat", "Sandal", "Shirt", "Sneaker", "Bag", "Ankle boot"])
    plt.show()
